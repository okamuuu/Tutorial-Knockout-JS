Event Binding

Run it:

Mouse over me Details

Source code:

  <p>
    <span data-bind="event: { mouseover: enableDetails, mouseout: disableDetails }">
      Mouse over me
    </span>
    <span data-bind="visible: detailsEnabled">
      <em>Details</em>
    </span>
  </p>

  <script>
  var viewModel = {
      detailsEnabled: ko.observable(false),
      enableDetails: function() {
          this.detailsEnabled(true);
      },
      disableDetails: function() {
          this.detailsEnabled(false);
      }
  }; 
  ko.applyBindings(viewModel);
  
  </script>
